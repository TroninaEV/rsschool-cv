    classPoints.push(yourPoints);
    let length = classPoints.length;
    let sum = 0;
    for (let i = 0; i < length; i++)
    {
        sum = sum + classPoints[i];
    }
    let average = sum/length;
    if(yourPoints > average){
        return true;
    }
        return false;
}
